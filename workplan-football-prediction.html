★ scoring_rules.js — Cấu hình rules tính điểm
📌 Quy tắc quan trọng

File này là nguồn sự thật duy nhất cho rules tính điểm. mod_scoring.js đọc từ đây — không hardcode điểm ở bất kỳ đâu khác. Muốn đổi rules: sửa file này → commit → GitHub Actions deploy tự động (~1 phút).

// scoring_rules.js — SỬA FILE NÀY để thay đổi rules tính điểm const SCORING_RULES = { GROUP: { correct_pts: 0, // đoán đúng W/D/L → không được điểm wrong_pts: -1, // đoán sai → trừ 1 điểm allow_draw: true, // có nút Hòa note: 'Vòng bảng' }, KNOCKOUT: { correct_pts: 0, // đoán đúng W/L → không được điểm wrong_pts: -1, // đoán sai → trừ 1 điểm allow_draw: false, // không có nút Hòa note: 'Loại trực tiếp' } }

Logic tính điểm — đọc từ scoring_rules.js, không hardcode
// mod_scoring.js // Mọi rules đến từ SCORING_RULES (scoring_rules.js) — không hardcode số nào ở đây async function calculateScore(userId) { const predictions = await sheets.getPredictions(userId) const results = await sheets.getResults() // match_id → { outcome, phase } let score = 0, correct = 0, wrong = 0, pending = 0 for (const pred of predictions) { const match = results[pred.match_id] if (!match?.outcome) { pending++; continue } const rule = SCORING_RULES[match.phase] // ← từ scoring_rules.js if (pred.prediction === match.outcome) { score += rule.correct_pts correct++ } else { score += rule.wrong_pts wrong++ } } return { score, correct, wrong, pending } }
